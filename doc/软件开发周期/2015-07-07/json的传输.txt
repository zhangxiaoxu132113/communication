alert(data);
/*
* 利用eval()函数转换成json数据格式，传入eval()函数的内容还是文本内容
* 		* 利用eval()函数进行转换json数据格式，需要利用"()"将传入的参数进行包裹
* 			* 如果不利用"()"进行包裹，传入的内容是"{}"，这时eval()函数会将其识别成"空的语句块"
* 			* 如果利用"()"进行包裹，传入的内容是"{}"，这时eval()函数会强行将其转换成json格式
* 
* 		* 加上"()"一定转换成功
*/
var json = eval("("+data+")");
